def f(m,n):
    return (-1)^(n+m)*factorial(n)*(sum([(-1)^j*binomial(n+m-j,n)*zeta(j,hold=True) for j in (2..m)])-binomial(n+m,m)+sum([binomial(n+m-j,m-1)*zeta(j,hold=True) for j in (2..n+1)]))
